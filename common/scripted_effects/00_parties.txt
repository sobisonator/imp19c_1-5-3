remove_party_leadership_effect = {
	if = {
		limit = {
			is_party_leader = yes
		}
		if = {
			limit = {
				employer = {
					is_republic = yes
					has_party = party:reactionary_party
				}
				party = reactionary_party
			}
			employer = {
				remove_party_leadership = reactionary_party
			}
		}
		else_if = {
			limit = {
				employer = {
					is_republic = yes
