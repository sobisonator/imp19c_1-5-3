local_wealth_pool_pulse = {
    effect = {
        if = {
            limit = {
                NOT = {
                    has_variable = local_wealth_pool_update
                }
            }
            set_variable = {
                name = local_wealth_pool_update
                value = 0
            }
        }
        change_variable = {
            name = "local_wealth_pool_update"
            add = local_wealth_pool_gain
        }
        change_variable = {
            name = "local_wealth_pool_update"
            subtract = local_wealth_pool_demanded
        }
    }
}
